70  template<typename ResultType>
71  xml_node<> * create_testcase_node(const std::shared_ptr<ResultType>& result){
72  xml_node<> * current_test_case_node = doc.allocate_node(node_type::node_element, "testcase");
73  const char * time_taken_cstr = to_string(result->get_time_taken()).c_str();
74  current_test_case_node->append_attribute(doc.allocate_attribute("time", allocate_for_doc(time_taken_cstr)));
75  current_test_case_node->append_attribute(doc.allocate_attribute("name", allocate_for_doc(result->get_testname().c_str())));
76  return current_test_case_node;
77  }

81  std::string run(ResultSetType resultset)
82  {
83  xml_node<> * node = doc.allocate_node(node_type::node_element, "testsuite");
84  doc.append_node(node);
85  node->append_attribute(doc.allocate_attribute("failures", allocate_for_doc(to_string(resultset.getFails().size()).c_str())));
86  node->append_attribute(doc.allocate_attribute("time", allocate_for_doc(to_string(resultset.getTotalExecutionTimeInSeconds()).c_str())));
87  node->append_attribute(doc.allocate_attribute("errors", allocate_for_doc(to_string(resultset.getAborts().size()).c_str())));
88  node->append_attribute(doc.allocate_attribute("skipped", allocate_for_doc(to_string(resultset.getIgnores().size()).c_str())));
89  node->append_attribute(doc.allocate_attribute("tests", allocate_for_doc(to_string(resultset.size()).c_str())));
90  node->append_attribute(doc.allocate_attribute("name", "C++ LTest Testsuite")); // TODO request from user or infer testsuite name
91  for(const auto& result : resultset.getIgnores()){
92  xml_node<> * current_test_case_node = create_testcase_node(result);
93  current_test_case_node->append_node(doc.allocate_node(node_type::node_element, "skipped"));
94  node->append_node(current_test_case_node);
95  }
96  for(const auto& result : resultset.getOK()){
97  xml_node<> * current_test_case_node = create_testcase_node(result);
98  node->append_node(current_test_case_node);
99  create_sysout_and_syserr_nodes(result, current_test_case_node);
100  }
101  for(const auto& result : resultset.getFails()){
102  xml_node<> * current_test_case_node = create_testcase_node(result);
103  xml_node<> * failure_node = doc.allocate_node(node_type::node_element, "failure", allocate_for_doc(result->getMessage().c_str()));
104  current_test_case_node->append_node(failure_node);
105  node->append_node(current_test_case_node);
106  create_sysout_and_syserr_nodes(result, current_test_case_node);
107  }
108  for(const auto& result : resultset.getAborts()){
109  xml_node<> * current_test_case_node = create_testcase_node(result);
110  xml_node<> * error_node = doc.allocate_node(node_type::node_element, "error", allocate_for_doc(result->getMessage().c_str()));
111  current_test_case_node->append_node(error_node);
112  node->append_node(current_test_case_node);
113  create_sysout_and_syserr_nodes(result, current_test_case_node);
114  }
115  std::stringstream output_stream;
116  output_stream << doc;
117  return output_stream.str();
118  }

120  };
